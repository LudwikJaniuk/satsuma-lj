38  public static double GetTourCost<TNode>(IEnumerable<TNode> tour, Func<TNode, TNode, double> cost)
39  {
40  double result = 0;
41  if (tour.Any())
42  {
43  TNode prev = tour.First();
44  foreach (var node in tour.Skip(1))

46  result += cost(prev, node);

48  }
49  }

92  private void Run()
93  {
94  // create a complete graph and run Kruskal with maximum degree constraint 2
95  CompleteGraph graph = new CompleteGraph(Nodes.Count, Directedness.Undirected);
96  Func<Arc, double> arcCost = (arc => Cost(Nodes[graph.GetNodeIndex(graph.U(arc))],
97  Nodes[graph.GetNodeIndex(graph.V(arc))]));
98  Kruskal<double> kruskal = new Kruskal<double>(graph, arcCost, _ => 2);
99  kruskal.Run();
100 
101  Dictionary<Node, Arc> firstArc = new Dictionary<Node, Arc>();
102  Dictionary<Node, Arc> secondArc = new Dictionary<Node, Arc>();
103  foreach (var arc in kruskal.Forest)
104  {
105  var u = graph.U(arc);
106  (firstArc.ContainsKey(u) ? secondArc : firstArc)[u] = arc;
107  var v = graph.V(arc);
108  (firstArc.ContainsKey(v) ? secondArc : firstArc)[v] = arc;
109  }
110 
111  foreach (var startNode in graph.Nodes())
112  {
113  if (kruskal.Degree[startNode] == 1)
114  {
115  Arc prevArc = Arc.Invalid;
116  Node n = startNode;
117  while (true)
118  {
119  tour.Add(Nodes[graph.GetNodeIndex(n)]);
120  if (prevArc != Arc.Invalid && kruskal.Degree[n] == 1) break;
121  Arc arc1 = firstArc[n];
122  prevArc = (arc1 != prevArc ? arc1 : secondArc[n]);
123  n = graph.Other(prevArc, n);
124  }
125  tour.Add(Nodes[graph.GetNodeIndex(startNode)]);
126  break;
127  }
128  }
129 
130  TourCost = TspUtils.GetTourCost(tour, Cost);
131  }
132  }

222  public bool Insert(TNode node)
223  {
224  if (!insertableNodes.Contains(node)) return false;
225  insertableNodes.Remove(node);
226  insertableNodeQueue.Remove(node);
227 
228  // find the optimal insertion place
229  LinkedListNode<TNode> insertAfter = null;
230  double bestIncrease = double.PositiveInfinity;
231  for (var llnode = tour.First; llnode != tour.Last; llnode = llnode.Next)
232  {
233  var llnode2 = llnode.Next;
234  double increase = Cost(llnode.Value, node) + Cost(node, llnode2.Value);
235  if (!llnode.Value.Equals(llnode2.Value))
236  increase -= Cost(llnode.Value, llnode2.Value);
237  if (increase < bestIncrease)
238  {
239  bestIncrease = increase;
240  insertAfter = llnode;
241  }
242  }
243 
244  LinkedListNode<TNode> llnodeNew = tourNodes[node] = tour.AddAfter(insertAfter, node);
245  TourCost += bestIncrease;
246 
247  // update distances
248  foreach (var n in insertableNodes)
249  {
250  double newPriority = PriorityFromCost(Cost(node, n));
251  if (newPriority < insertableNodeQueue[n]) insertableNodeQueue[n] = newPriority;
252  }
253 
254  return true;
255  }

302  public bool Step()
303  {
304  bool improved = false;
305  for (int i = 0; i < tour.Count-3; i++) // first arc
306  {
307  for (int j = i + 2, jmax = tour.Count - (i == 0 ? 2 : 1); j < jmax; j++) // second arc
308  {
309  double increase = Cost(tour[i], tour[j]) + Cost(tour[i+1], tour[j+1]) -
310  (Cost(tour[i], tour[i+1]) + Cost(tour[j], tour[j+1]));
311  if (increase < 0)
312  {
313  TourCost += increase;
314  tour.Reverse(i + 1, j - i);
315  improved = true;
316  }
317  }
318  }
319  return improved;
320  }

353  private void Run()
354  {
355  Func<Node, Node, double> cost = (u, v) => (Graph.Arcs(u, v, ArcFilter.Forward).Any() ? 1 : 10);
356  IEnumerable<Node> tour = null;
357  double minimumTourCost = Graph.NodeCount();
358 
359  // Use the insertion tsp combined with 2-OPT heuristic.
360  InsertionTsp<Node> insertionTsp = new InsertionTsp<Node>(Graph.Nodes(), cost);
361  insertionTsp.Run();
362  if (insertionTsp.TourCost == minimumTourCost) tour = insertionTsp.Tour;
363  else
364  {
365  Opt2Tsp<Node> opt2Tsp = new Opt2Tsp<Node>(cost, insertionTsp.Tour, insertionTsp.TourCost);
366  opt2Tsp.Run();
367  if (opt2Tsp.TourCost == minimumTourCost) tour = opt2Tsp.Tour;
368  }
369 
370  // convert the tour (node sequence) into a path (arc sequence connecting two nodes)
371  if (tour == null) Cycle = null;
372  else
373  {
374  var cycle = new Path(Graph);
375  if (tour.Any())
376  {
377  Node prev = Node.Invalid;
378  foreach (var n in tour)
379  {
380  if (prev == Node.Invalid) cycle.Begin(n);
381  else cycle.AddLast(Graph.Arcs(prev, n, ArcFilter.Forward).First());
382 
383  prev = n;
384  }
385  cycle.AddLast(Graph.Arcs(prev, tour.First(), ArcFilter.Forward).First());
386  } // if tour is not empty
387  Cycle = cycle;
388  }
389  }
