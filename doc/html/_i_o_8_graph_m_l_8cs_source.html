191  public override void ReadData(XElement x, object key)
192  {
193  if (x == null)

202  if (key == null)
203  {
204  HasDefaultValue = true;
205  DefaultValue = value;
206  }
207  else Values[key] = value;

285  private static StandardType ParseType(Type t)
286  {
287  if (t == typeof(bool)) return StandardType.Bool;
288  if (t == typeof(double)) return StandardType.Double;
289  if (t == typeof(float)) return StandardType.Float;

621  public void Load(XDocument doc)
622  {
623  // Namespaces are ignored so we can load broken documents.
624  if (Graph == null) Graph = new CustomGraph();
625  IBuildableGraph buildableGraph = (IBuildableGraph)Graph;
626  buildableGraph.Clear();
627  XElement xGraphML = doc.Root;
628 
629  // load properties
630  Properties.Clear();
631  Dictionary<string, GraphMLProperty> propertyById = new Dictionary<string, GraphMLProperty>();
632  foreach (var xKey in Utils.ElementsLocal(xGraphML, "key"))
633  {
634  foreach (var handler in PropertyLoaders)
635  {
636  try
637  {
638  GraphMLProperty p = handler(xKey);
639  Properties.Add(p);
640  propertyById[p.Id] = p;
641  break;
642  }
643  catch (ArgumentException) { }
644  }
645  }
646 
647  // load graph
648  XElement xGraph = Utils.ElementLocal(xGraphML, "graph");
649  Directedness defaultDirectedness = (xGraph.Attribute("edgedefault").Value == "directed" ?
650  Directedness.Directed : Directedness.Undirected);
651  ReadProperties(propertyById, xGraph, Graph);
652  // load nodes
653  NodeId = new Dictionary<Node, string>();
654  Dictionary<string, Node> nodeById = new Dictionary<string, Node>();
655  foreach (var xNode in Utils.ElementsLocal(xGraph, "node"))
656  {
657  Node node = buildableGraph.AddNode();
658  string id = xNode.Attribute("id").Value;
659  NodeId[node] = id;
660  nodeById[id] = node;
661  ReadProperties(propertyById, xNode, node);
662  }
663  // load arcs
664  ArcId = new Dictionary<Arc, string>();
665  foreach (var xArc in Utils.ElementsLocal(xGraph, "edge"))
666  {
667  Node u = nodeById[xArc.Attribute("source").Value];
668  Node v = nodeById[xArc.Attribute("target").Value];
669 
670  Directedness dir = defaultDirectedness;
671  XAttribute dirAttr = xArc.Attribute("directed");
672  if (dirAttr != null) dir = (dirAttr.Value == "true" ? Directedness.Directed : Directedness.Undirected);
673 
674  Arc arc = buildableGraph.AddArc(u, v, dir);
675  XAttribute xId = xArc.Attribute("id");
676  if (xId != null)
677  ArcId[arc] = xId.Value;
678  ReadProperties(propertyById, xArc, arc);
679  }
680  }

745  private void Save(XmlWriter xml)
746  {
747  xml.WriteStartDocument();
748  xml.WriteStartElement("graphml", xmlns.NamespaceName);
749  xml.WriteAttributeString("xmlns", "xsi", null, xmlnsXsi.NamespaceName);
750  xml.WriteAttributeString("xmlns", "y", null, xmlnsY.NamespaceName);
751  xml.WriteAttributeString("xmlns", "yed", null, xmlnsYed.NamespaceName);
752  xml.WriteAttributeString("xsi", "schemaLocation", null, xsiSchemaLocation);
753 
754  for (int i = 0; i < Properties.Count; i++)
755  {
756  var p = Properties[i];
757  p.Id = "d" + i;
758  p.GetKeyElement().WriteTo(xml);
759  }
760 
761  xml.WriteStartElement("graph", xmlns.NamespaceName);
762  xml.WriteAttributeString("id", "G");
763  xml.WriteAttributeString("edgedefault", "directed");
764  xml.WriteAttributeString("parse.nodes", Graph.NodeCount().ToString(CultureInfo.InvariantCulture));
765  xml.WriteAttributeString("parse.edges", Graph.ArcCount().ToString(CultureInfo.InvariantCulture));
766  xml.WriteAttributeString("parse.order", "nodesfirst");
767  DefinePropertyValues(xml, Graph);
768 
769  Dictionary<string, Node> nodeById = new Dictionary<string, Node>();
770  if (NodeId == null)
771  NodeId = new Dictionary<Node,string>();
772  foreach (var kv in NodeId)
773  {
774  if (nodeById.ContainsKey(kv.Value))
775  throw new Exception("Duplicate node id " + kv.Value);
776  nodeById[kv.Value] = kv.Key;
777  }
778  foreach (var node in Graph.Nodes())
779  {
780  string id;
781  NodeId.TryGetValue(node, out id);
782  if (id == null)
783  {
784  id = node.Id.ToString(CultureInfo.InvariantCulture);
785  while (nodeById.ContainsKey(id))
786  id += '_';
787  NodeId[node] = id;
788  nodeById[id] = node;
789  }
790 
791  xml.WriteStartElement("node", xmlns.NamespaceName);
792  xml.WriteAttributeString("id", id);
793  DefinePropertyValues(xml, node);
794  xml.WriteEndElement(); // node
795  }
796 
797  foreach (var arc in Graph.Arcs())
798  {
799  string id;
800  if (ArcId != null)
801  ArcId.TryGetValue(arc, out id);
802  else
803  id = null;
804 
805  xml.WriteStartElement("edge", xmlns.NamespaceName);
806  if (id != null) xml.WriteAttributeString("id", id);
807  if (Graph.IsEdge(arc)) xml.WriteAttributeString("directed", "false");
808  xml.WriteAttributeString("source", NodeId[Graph.U(arc)]);
809  xml.WriteAttributeString("target", NodeId[Graph.V(arc)]);
810  DefinePropertyValues(xml, arc);
811  xml.WriteEndElement(); // edge
812  }
813 
814  xml.WriteEndElement(); // graph
815  xml.WriteEndElement(); // graphml
816  }
